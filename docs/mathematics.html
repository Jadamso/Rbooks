3.3 Functions

Function of a vector

## Add two to any vector
add2 <- function(x1) {
    x1+2
}
add2(x)
## [1]  2  3  5 12  8
## Generalization
addn <- function(x1,n=2) {
    x1+n
}
addn(x)
## [1]  2  3  5 12  8
addn(x,3)
## [1]  3  4  6 13  9

Function for two vectors

sum_squared <- function(x1, x2) {
    y <- (x1 + x2)^2
    return(y)
}

sum_squared(1, 3)
## [1] 16
sum_squared(x, 2)
## [1]   4   9  25 144  64
sum_squared(x, NA) 
## [1] NA NA NA NA NA
sum_squared(x, x)
## [1]   0   4  36 400 144
sum_squared(x, 2*x)
## [1]   0   9  81 900 324

Apply extends to arrays

apply(a, 1, mean)    # Row means
## [1] 12 13
apply(a, 2, mean)    # Column means
## [1] 10.5 12.5 14.5
apply(a, 3, mean)    # "Layer" means
## [1]  3.5  9.5 15.5 21.5
apply(a, 1:2, mean)  # Row/Column combination 
##      [,1] [,2] [,3]
## [1,]   10   12   14
## [2,]   11   13   15
