6.2 Data processing simulation

Make random datasets

make_noisy_data <- function(n){
    x <- seq(1,10, length.out=n)
    e <- rnorm(length(x), mean=0, sd=10)
    y <- .25*x + e
    xy_mat <- data.frame(ID=seq(x),x=x, y=y)
    return(xy_mat)
}
dat1 <- make_noisy_data(6)
